import { CommonModule } from '@angular/common';
import { Component, OnInit, OnDestroy } from '@angular/core';
import { FormsModule } from '@angular/forms';
import {  ElementRef, ViewChild } from '@angular/core';
//import * as Stomp from 'stompjs';
import { Client, Stomp ,IMessage } from '@stomp/stompjs';
//import * as SockJS from 'sockjs-client';

import SockJS from 'sockjs-client'
@Component({
  selector: 'app-multiple-call',
  standalone: true,
  imports: [FormsModule,CommonModule ],
  templateUrl: './multiple-call.component.html',
  styleUrl: './multiple-call.component.scss'
})
export class MultipleCallComponent implements  OnInit  {

@ViewChild('localVideo') localVideo!: ElementRef<HTMLVideoElement>;
  peerConnections: { [id: string]: RTCPeerConnection } = {};
  localStream!: MediaStream;
  stompClient!: Client;
  myId = Math.random().toString(36).substring(2);
  roomId = 'room1';

  async ngOnInit() {
    await this.startLocalStream();
    this.connectSignaling();
  }

  async startLocalStream() {
    this.localStream = await navigator.mediaDevices.getUserMedia({
      video: true,
      audio: true,
    });
    this.localVideo.nativeElement.srcObject = this.localStream;
  }

  connectSignaling() {
    this.stompClient = new Client({
      // Use SockJS for fallback, since Spring Boot uses SockJS
      webSocketFactory: () => new SockJS('http://localhost:8080/ws'),
      reconnectDelay: 5000,
      debug: (str) => {
        console.log('[STOMP]', str);
      },
      onConnect: (frame) => {
        // Subscribe to receive signaling messages directed to this user
        this.stompClient.subscribe('/user/queue/signal', (msg: IMessage) => {
          const message: SignalMessage = JSON.parse(msg.body);
          console.log('Received signal:', message);
          this.handleSignal(message);
        });

        // Subscribe to get current participants in the room
        this.stompClient.subscribe('/user/queue/participants', (msg: IMessage) => {
          const participants: string[] = JSON.parse(msg.body).participants;
          participants.forEach((peerId) => {
            if (peerId !== this.myId) {
              this.createPeerConnection(peerId, true);
            }
          });
        });

        // Subscribe to notification when new peers join
        this.stompClient.subscribe('/user/queue/new-peer', (msg: IMessage) => {
          const newPeerId = JSON.parse(msg.body).newPeerId;
          if (newPeerId !== this.myId) {
            this.createPeerConnection(newPeerId, false);
          }
        });

        // Notify server that we joined
        this.sendSignal({
          type: 'join',
          from: this.myId,
          to: '',
          roomId: this.roomId,
        });
      },
    });

    this.stompClient.activate();
  }

  async handleSignal(message: SignalMessage) {
    const peerId = message.from;
    switch (message.type) {
      case 'offer':
        await this.createPeerConnection(peerId, false, message.data);
        break;
      case 'answer':
        await this.peerConnections[peerId].setRemoteDescription(
          new RTCSessionDescription(message.data)
        );
        break;
      case 'candidate':
        await this.peerConnections[peerId].addIceCandidate(
          new RTCIceCandidate(message.data)
        );
        break;
    }
  }

  async createPeerConnection(
    peerId: string,
    isInitiator: boolean,
    remoteOffer?: any
  ) {
    if (this.peerConnections[peerId]) return;
 console.log(`Creating peer connection to ${peerId}, initiator: ${isInitiator}`);
    const pc = new RTCPeerConnection({
      iceServers: [{ urls: 'stun:stun.l.google.com:19302' }],
    });

    this.localStream.getTracks().forEach((track) => pc.addTrack(track, this.localStream));

    pc.onicecandidate = (event) => {
      if (event.candidate) {
          console.log(`Sending ICE candidate to ${peerId}`, event.candidate);
        this.sendSignal({
          type: 'candidate',
          from: this.myId,
          to: peerId,
          roomId: this.roomId,
          data: event.candidate,
        });
      }
    };
    console.log('STOMP connected');
console.log('Sending join for room:', this.roomId);
//console.log('Received signal:', message);
console.log(`Peer connection for ${peerId} created`);

    pc.ontrack = (event) => {
      console.log(`Received remote stream from ${peerId}`, event.streams[0]);
      this.addRemoteVideo(peerId, event.streams[0]);
    };

    this.peerConnections[peerId] = pc;

    if (isInitiator) {
      const offer = await pc.createOffer();
      await pc.setLocalDescription(offer);
      this.sendSignal({
        type: 'offer',
        from: this.myId,
        to: peerId,
        roomId: this.roomId,
        data: offer,
      });
    } else if (remoteOffer) {
      await pc.setRemoteDescription(new RTCSessionDescription(remoteOffer));
      const answer = await pc.createAnswer();
      await pc.setLocalDescription(answer);
      this.sendSignal({
        type: 'answer',
        from: this.myId,
        to: peerId,
        roomId: this.roomId,
        data: answer,
      });
    }
  }

  addRemoteVideo(peerId: string, stream: MediaStream) {
    if (document.getElementById(peerId)) return;
console.log(`Adding remote video element for ${peerId}`);
    const video = document.createElement('video');
    video.id = peerId;
    video.autoplay = true;
    video.playsInline = true;
    video.srcObject = stream;
    video.width = 300;
    video.height = 200;
    video.style.margin = '5px';

    document.getElementById('remoteVideos')?.appendChild(video);
  }

  sendSignal(message: SignalMessage) {
    this.stompClient.publish({
      destination: '/app/signal',
      body: JSON.stringify(message),
    });
  }
}

interface SignalMessage {
  type: string;
  from: string;
  to: string;
  roomId: string;
  data?: any;
}
